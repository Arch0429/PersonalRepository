```js title:"즉시실행함수" hl:14-18
// let은 다르게 동작하지만
// ES6 문법을 사용하지 못하는 경우 즉시실행 함수를 사용한다.
function printNumber() {
	for(var i=0 ; i<4 ; i++) {
		(function(j){
			setTimeout(function(){
				console.log(j)
			}, j * 1000)
		})(i)
	}
}

// 결과
// 0 1 2 3

printNumber();
```

```js title:"Promise 문법" hl:28-29
// Promise 는 new 키워드를 사용해서 생성할 수 있다.
// 성공 시, resolve 를 실패 시 reject

new Promise(function(resolve, reject) {
	if(success) {
		resolve()
	} else {
		reject()
	}
})

// 사용 예시
function myFunc(username) {
	return new Promise((resolve, reject) => {
		if(username) {
			resolve(username)
		}
	})
}

myFunc("Kim").then((username) => {
	console.log(username) // Kim
})

// 위처럼 resolve 를 통해서 전달한 파라미터는 프로미스의 then API 의 파라미터로
// 전달되며 reject 를 통해서 전달한 파라미터는 catch API 로 전달된다.
// 추가로 성공여부에 관계없이 특정 동작을 지정하려면 finaly API 를 사용한다.
// resolve 나 reject 에 여러개의 파라미터를 전달해도 하나만 전달되기 때문에
// 만약 여러개의 파라미터를 전달하려면 Object 등을 활용해야한다.

var data = {
	"username" : "Kim",
	"age" : 29
	...
}

resolve(data);
```
