```js title:"즉시실행함수" hl:14-18
// let은 다르게 동작하지만
// ES6 문법을 사용하지 못하는 경우 즉시실행 함수를 사용한다.
function printNumber() {
	for(var i=0 ; i<4 ; i++) {
		(function(j){
			setTimeout(function(){
				console.log(j)
			}, j * 1000)
		})(i)
	}
}

// 결과
// 0 1 2 3

printNumber();
```
